var enlacesSvg = document.querySelectorAll(".country");



// Agregar un event listener a cada elemento <path>
enlacesSvg.forEach(function (enlaceSvg) {
    enlaceSvg.addEventListener("click", function (event) {
        // Detener el comportamiento predeterminado del enlace
        event.preventDefault();
        // Obtener el valor del atributo xlink:title del elemento <path> actual
        var titulo = enlaceSvg.getAttribute("xlink:title");
        // Mostrar el valor del título en la consola (puedes hacer lo que quieras con este valor)
        console.log(titulo);

        var xhttp = new XMLHttpRequest();

        // Definir la función de retorno de llamada cuando la solicitud se complete
        xhttp.onreadystatechange = function () {
            if (this.readyState == 4 && this.status == 200) {
                // La solicitud se ha completado y la respuesta del servidor está lista
                try {
                    console.log(this.responseText); // Muestra la respuesta del servidor en la consola del navegador
                } catch (error) {
                    console.error("Error:", error);
                }
            }
        };

        // Abrir una nueva solicitud AJAX
        xhttp.open("POST", "apiconsumer.php", true);

        // Establecer el encabezado de la solicitud para indicar que se enviarán datos de formulario
        xhttp.setRequestHeader("Content-type", "application/x-www-form-urlencoded");

        // Enviar la solicitud con los datos que deseas enviar al servidor
        xhttp.send(titulo);
    });
});
